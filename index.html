function digPow(n, p){
    let temp = 0;
    let value = n.toString().split("");
    for (let i = 0; i < value.length; i++) {
        temp += Math.pow(parseFloat(value[i]), (p + i));
    }
    return temp % n == 0 ? temp / n : -1;
}
